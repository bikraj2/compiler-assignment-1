set int big;  # declaring an integer variable

func fibonacci ( int n; int)  # comment
<
    int a = 0;   # first Fibonacci number
    int b = 1;   # second Fibonacci number
    int next;    # to hold the next Fibonacci number

       loop (int i = 2; i <= n; i = i + 1) :
    <
        next = a + b;  # calculate the next Fibonacci number
        a = b;         # move to the next number in the sequence
        b = next;      # update b to the next number
    >;

    return b;  # return the nth Fibonacci number
>

func factorial(int n; int)  # comment
<
    int fact = 1;  # Initialize factorial result

    loop (int i = 1; i <= n; i = i + 1):
    <
        fact = fact * i;  # Multiply the result by i
    >;

    return fact;  # return the factorial of n
>

int n = 5; 
# Example input

# Call and print the results of both functions
print(fibonacci(n));  # call the Fibonacci function
print(factorial(n));  # call the Factorial function
